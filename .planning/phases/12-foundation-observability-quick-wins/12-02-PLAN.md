```typescript
import { Injectable, OnModuleInit, OnModuleDestroy } from '@nestjs/common';
import { PrismaClient } from '@prisma/client';
import { createSoftDeleteExtension } from 'prisma-extension-soft-delete';

@Injectable()
export class PrismaService extends PrismaClient implements OnModuleInit, OnModuleDestroy {
  constructor() {
    super();
    // Apply soft-delete extension — all findMany/findFirst/findUnique
    // on these models will auto-filter deletedAt IS NOT NULL.
    // delete() calls become update({ deletedAt: new Date() }).
    const softDeleteModels = {
      User: true,
      Fabric: true,
      Supplier: true,
      Customer: true,
      Product: true,
      ProductBundle: true,
    };
    const modelConfig = Object.fromEntries(
      Object.keys(softDeleteModels).map((model) => [
        model,
        { field: 'deletedAt', createValue: (deleted: boolean) => (deleted ? new Date() : null) },
      ]),
    );
    // Note: $extends returns a new type. Since PrismaService extends PrismaClient,
    // the extension modifies the internal client behavior through Prisma's proxy mechanism.
    // The returned extended client's methods are available on `this` because $extends
    // patches the prototype chain.
    this.$extends(createSoftDeleteExtension({ models: modelConfig }));
  }

  async onModuleInit() {
    await this.$connect();
  }

  async onModuleDestroy() {
    await this.$disconnect();
  }
}
```
