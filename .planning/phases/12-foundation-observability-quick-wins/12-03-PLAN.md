/**
 * Normalize cell value for header matching:
 * - Extract plain text from RichText objects
 * - Trim whitespace

function normalizeHeaderValue(cell: ExcelJS.Cell): string {
  const value = cell.value;
  if (value === null || value === undefined) return '';
  if (typeof value === 'object' && 'richText' in value) {
    // RichText: extract concatenated plain text
    return (value.richText as Array<{ text: string }>)
      .map((rt) => rt.text)
      .join('')
      .trim()
      .toLowerCase();
  }
  return String(value).trim().toLowerCase();
}

**1. Add `mapApiErrorsToFormFields` to `frontend/src/utils/errorMessages.ts`** per D-14:

```typescript
import type { FieldData } from 'antd/es/form/interface';

/**
 * Map backend validation error response to Ant Design Form field errors.
 * Handles both single-message and array-message formats from class-validator.
 *
 * Backend error.message can be:
 * - string: "companyName should not be empty"
 * - string[]: ["companyName should not be empty", "email must be an email"]
 *
 * @param apiError - The ApiError object
 * @returns FieldData[] for use with form.setFields()
 */
export function mapApiErrorsToFormFields(
  apiError: ApiError
): FieldData[] {
  if (!apiError?.message) return [];

  const fields: FieldData[] = [];
  const messages = Array.isArray(apiError.message)
    ? apiError.message
    : [apiError.message];

  for (const msg of messages) {
    if (typeof msg !== 'string') continue;
    const parsed = parseFieldError(msg);
    if (parsed) {
      fields.push({ name: parsed.field, errors: [parsed.message] });
    }
  }
  return fields;
}
```

Note: The `ApiError` type has `message: string`, but NestJS ValidationPipe can return `message: string[]`. The function handles both by type-checking at runtime. If TypeScript complains about `Array.isArray(apiError.message)`, cast `apiError.message as string | string[]` since the runtime type can be either.
