4. Search function using `Promise.allSettled` — fetches products/fabrics in parallel, then resolves lowest-price suppliers:
```typescript
async function unifiedSearch(keyword: string): Promise<UnifiedSearchResult[]> {
  // Phase 1: Fetch fabrics and products in parallel
  const [fabricResult, productResult] = await Promise.allSettled([
    getFabrics({ keyword, pageSize: 10 }),
    getProducts({ keyword, pageSize: 10 }),
  ]);

  const results: UnifiedSearchResult[] = [];
  const fabricItems = fabricResult.status === 'fulfilled' ? fabricResult.value.items : [];
  const productItems = productResult.status === 'fulfilled' ? productResult.value.items : [];

  // Phase 2: Fetch suppliers for all results in parallel (lowest-price resolution)
  // For each fabric/product, query its suppliers sorted by purchasePrice (asc), take first.
  // Use Promise.allSettled so one failure doesn't block others.
  const supplierPromises: Promise<{
    type: 'fabric' | 'product';
    itemId: number;
    supplierId: number | null;
    price: number | null;
  }>[] = [];

  for (const fabric of fabricItems) {
    supplierPromises.push(
      getFabricSuppliers(fabric.id, { pageSize: 1, sortBy: 'purchasePrice', sortOrder: 'asc' })
        .then((res) => ({
          type: 'fabric' as const,
          itemId: fabric.id,
          supplierId: res.items[0]?.supplierId ?? null,
          price: res.items[0] ? Number(res.items[0].purchasePrice) : null,
        }))
        .catch(() => ({
          type: 'fabric' as const,
          itemId: fabric.id,
          supplierId: null,
          price: null,
        }))
    );
  }

  for (const product of productItems) {
    supplierPromises.push(
      getProductSuppliers(product.id, { pageSize: 1, sortBy: 'purchasePrice', sortOrder: 'asc' })
        .then((res) => ({
          type: 'product' as const,
          itemId: product.id,
          supplierId: res.items[0]?.supplierId ?? null,
          price: res.items[0] ? Number(res.items[0].purchasePrice) : null,
        }))
        .catch(() => ({
          type: 'product' as const,
          itemId: product.id,
          supplierId: null,
          price: null,
        }))
    );
  }

  const supplierResults = await Promise.all(supplierPromises);

  // Build supplier lookup maps
  const fabricSupplierMap = new Map<number, { supplierId: number | null; price: number | null }>();
  const productSupplierMap = new Map<number, { supplierId: number | null; price: number | null }>();
  for (const sr of supplierResults) {
    if (sr.type === 'fabric') {
      fabricSupplierMap.set(sr.itemId, { supplierId: sr.supplierId, price: sr.price });
    } else {
      productSupplierMap.set(sr.itemId, { supplierId: sr.supplierId, price: sr.price });
    }
  }

  // Phase 3: Assemble results with supplier data
  for (const fabric of fabricItems) {
    const supplierInfo = fabricSupplierMap.get(fabric.id);
    results.push({
      type: 'fabric',
      id: fabric.id,
      code: fabric.fabricCode,
      name: fabric.name,
      categoryLabel: '面料',
      defaultPrice: fabric.defaultPrice,
      unit: '米',
      lowestSupplierId: supplierInfo?.supplierId ?? null,
      lowestSupplierPrice: supplierInfo?.price ?? null,
      compositeValue: `fabric:${fabric.id}`,
    });
  }

  for (const product of productItems) {
    const supplierInfo = productSupplierMap.get(product.id);
    results.push({
      type: 'product',
      id: product.id,
      code: product.productCode,
      name: product.name,
      subCategory: product.subCategory,
      categoryLabel: CATEGORY_TAG_LABELS[product.subCategory] || product.subCategory,
      defaultPrice: product.defaultPrice,
      unit: UNIT_BY_SUB_CATEGORY[product.subCategory] || '个',
      lowestSupplierId: supplierInfo?.supplierId ?? null,
      lowestSupplierPrice: supplierInfo?.price ?? null,
      compositeValue: `product:${product.id}`,
    });
  }

  return results;
}
```

**IMPORTANT implementation note:** The supplier API calls (`getFabricSuppliers`/`getProductSuppliers`) accept `sortBy` and `sortOrder` params via the existing QueryFabricSuppliersParams/QueryProductSuppliersParams. Check the actual query param names by reading the existing API usage. If the backend sorts by purchasePrice by default or does not support sort params, just fetch with `pageSize: 1` and the first result will be the cheapest (backend default sort may be by purchasePrice asc — verify by reading the service). If sort params are not available, fetch `pageSize: 5` and pick the min-price entry client-side.

export function parseCompositeValue(value: string): { type: 'fabric' | 'product'; id: number } | null {
  const [type, idStr] = value.split(':');
  const id = parseInt(idStr, 10);
  if ((type === 'fabric' || type === 'product') && !isNaN(id)) {
    return { type: type as 'fabric' | 'product', id };
  }
  return null;
}

```typescript
// OLD: Form.Item name={[fieldName, 'fabricId']} with FabricSelector
// NEW:
<Form.Item
  name={[fieldName, 'productSelection']}
  label="产品"
  rules={[{ required: true, message: '请选择产品' }]}
>
  <UnifiedProductSelector
    placeholder="搜索面料或产品"
    onChange={(compositeValue, result) => {
      if (result) {
        const parsed = parseCompositeValue(compositeValue!);
        if (parsed) {
          // Set fabricId or productId, clear the other
          if (parsed.type === 'fabric') {
            form.setFieldValue(['items', fieldName, 'fabricId'], parsed.id);
            form.setFieldValue(['items', fieldName, 'productId'], undefined);
          } else {
            form.setFieldValue(['items', fieldName, 'productId'], parsed.id);
            form.setFieldValue(['items', fieldName, 'fabricId'], undefined);
          }
          // Auto-populate unit
          form.setFieldValue(['items', fieldName, 'unit'], result.unit);
          // Auto-populate supplier (lowest-price from supplier relationship)
          if (result.lowestSupplierId != null) {
            form.setFieldValue(['items', fieldName, 'supplierId'], result.lowestSupplierId);
          }
          // Auto-populate purchase price from lowest supplier price, fallback to defaultPrice
          const autoPrice = result.lowestSupplierPrice ?? result.defaultPrice;
          if (autoPrice != null) {
            form.setFieldValue(['items', fieldName, 'purchasePrice'], autoPrice);
          }
        }
      } else {
        // Cleared selection
        form.setFieldValue(['items', fieldName, 'fabricId'], undefined);
        form.setFieldValue(['items', fieldName, 'productId'], undefined);
        form.setFieldValue(['items', fieldName, 'unit'], undefined);
        form.setFieldValue(['items', fieldName, 'supplierId'], undefined);
        form.setFieldValue(['items', fieldName, 'purchasePrice'], undefined);
      }
    }}
  />
</Form.Item>
```
