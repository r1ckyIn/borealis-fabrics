4. Search function using `Promise.allSettled`:
```typescript
async function unifiedSearch(keyword: string): Promise<UnifiedSearchResult[]> {
  const [fabricResult, productResult] = await Promise.allSettled([
    getFabrics({ keyword, pageSize: 10 }),
    getProducts({ keyword, pageSize: 10 }),
  ]);

  const results: UnifiedSearchResult[] = [];

  if (fabricResult.status === 'fulfilled') {
    fabricResult.value.items.forEach((fabric) => {
      results.push({
        type: 'fabric',
        id: fabric.id,
        code: fabric.fabricCode,
        name: fabric.name,
        categoryLabel: '面料',
        defaultPrice: fabric.defaultPrice,
        unit: '米',
        compositeValue: `fabric:${fabric.id}`,
      });
    });
  }

  if (productResult.status === 'fulfilled') {
    productResult.value.items.forEach((product) => {
      results.push({
        type: 'product',
        id: product.id,
        code: product.productCode,
        name: product.name,
        subCategory: product.subCategory,
        categoryLabel: CATEGORY_TAG_LABELS[product.subCategory] || product.subCategory,
        defaultPrice: product.defaultPrice,
        unit: UNIT_BY_SUB_CATEGORY[product.subCategory] || '个',
        compositeValue: `product:${product.id}`,
      });
    });
  }

  return results;
}
```

```typescript
export function parseCompositeValue(value: string): { type: 'fabric' | 'product'; id: number } | null {
  const [type, idStr] = value.split(':');
  const id = parseInt(idStr, 10);
  if ((type === 'fabric' || type === 'product') && !isNaN(id)) {
    return { type: type as 'fabric' | 'product', id };
  }
  return null;
}
```

3. Replace the FabricSelector Form.Item with UnifiedProductSelector:
```typescript
// OLD: Form.Item name={[fieldName, 'fabricId']} with FabricSelector
// NEW:
<Form.Item
  name={[fieldName, 'productSelection']}
  label="产品"
  rules={[{ required: true, message: '请选择产品' }]}
>
  <UnifiedProductSelector
    placeholder="搜索面料或产品"
    onChange={(compositeValue, result) => {
      if (result) {
        const parsed = parseCompositeValue(compositeValue!);
        if (parsed) {
          // Set fabricId or productId, clear the other
          if (parsed.type === 'fabric') {
            form.setFieldValue(['items', fieldName, 'fabricId'], parsed.id);
            form.setFieldValue(['items', fieldName, 'productId'], undefined);
          } else {
            form.setFieldValue(['items', fieldName, 'productId'], parsed.id);
            form.setFieldValue(['items', fieldName, 'fabricId'], undefined);
          }
          // Auto-populate unit
          form.setFieldValue(['items', fieldName, 'unit'], result.unit);
          // Auto-populate purchase price if available
          if (result.defaultPrice != null) {
            form.setFieldValue(['items', fieldName, 'purchasePrice'], result.defaultPrice);
          }
        }
      } else {
        // Cleared selection
        form.setFieldValue(['items', fieldName, 'fabricId'], undefined);
        form.setFieldValue(['items', fieldName, 'productId'], undefined);
        form.setFieldValue(['items', fieldName, 'unit'], undefined);
      }
    }}
  />
</Form.Item>
```
