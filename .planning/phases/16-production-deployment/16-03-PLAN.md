    Create `deploy/deploy.sh`:
    ```bash
    #!/bin/bash
    # Borealis Fabrics -- Production Deployment Script
    # Usage: ./deploy/deploy.sh [--skip-build] [--skip-migrate]
    set -euo pipefail

    SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
    PROJECT_DIR="$(dirname "$SCRIPT_DIR")"
    cd "$PROJECT_DIR"

    # Parse flags
    SKIP_BUILD=false
    SKIP_MIGRATE=false
    for arg in "$@"; do
      case $arg in
        --skip-build) SKIP_BUILD=true ;;
        --skip-migrate) SKIP_MIGRATE=true ;;
      esac
    done

    echo "=== Borealis Fabrics Deploy ==="
    echo "Project dir: $PROJECT_DIR"
    echo "Skip build: $SKIP_BUILD"
    echo "Skip migrate: $SKIP_MIGRATE"

    # 1. Pull latest code
    echo ""
    echo "--- Step 1: Pull latest code ---"
    git pull origin main

    # 2. Build frontend
    if [ "$SKIP_BUILD" = false ]; then
      echo ""
      echo "--- Step 2: Build frontend ---"
      cd frontend
      pnpm install --frozen-lockfile
      pnpm build
      cd "$PROJECT_DIR"

      # 3. Build and restart Docker containers
      echo ""
      echo "--- Step 3: Build Docker images ---"
      docker compose -f docker-compose.prod.yml build --no-cache
    fi

    # 4. Stop old containers
    echo ""
    echo "--- Step 4: Stop old containers ---"
    docker compose -f docker-compose.prod.yml down

    # 5. Start new containers
    echo ""
    echo "--- Step 5: Start new containers ---"
    docker compose -f docker-compose.prod.yml up -d

    # 6. Run Prisma migrations
    if [ "$SKIP_MIGRATE" = false ]; then
      echo ""
      echo "--- Step 6: Run database migrations ---"
      sleep 5  # Wait for NestJS container to be ready
      docker compose -f docker-compose.prod.yml exec nestjs npx prisma migrate deploy
    fi

    # 7. Health check
    echo ""
    echo "--- Step 7: Health check ---"
    sleep 5
    MAX_RETRIES=10
    RETRY=0
    until curl -sf http://localhost/health > /dev/null 2>&1; do
      RETRY=$((RETRY + 1))
      if [ $RETRY -ge $MAX_RETRIES ]; then
        echo "ERROR: Health check failed after $MAX_RETRIES retries"
        echo "Check logs: docker compose -f docker-compose.prod.yml logs nestjs"
        exit 1
      fi
      echo "Waiting for health check... ($RETRY/$MAX_RETRIES)"
      sleep 3
    done

    echo ""
    echo "=== Deploy Complete ==="
    echo "Health: $(curl -s http://localhost/health)"
    echo ""
    echo "Service status:"
    docker compose -f docker-compose.prod.yml ps
    ```

    Create `deploy/rollback.sh`:
    ```bash
    #!/bin/bash
    # Borealis Fabrics -- Rollback to Previous Deployment
    # Usage: ./deploy/rollback.sh [git-commit-sha]
    set -euo pipefail

    SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
    PROJECT_DIR="$(dirname "$SCRIPT_DIR")"
    cd "$PROJECT_DIR"

    COMMIT=${1:-""}

    echo "=== Borealis Fabrics Rollback ==="

    if [ -z "$COMMIT" ]; then
      echo "No commit SHA provided. Rolling back to previous commit."
      COMMIT=$(git log --oneline -2 | tail -1 | awk '{print $1}')
    fi

    echo "Rolling back to: $COMMIT"
    echo ""

    # 1. Checkout specific commit
    git checkout "$COMMIT"

    # 2. Rebuild frontend
    echo "--- Rebuilding frontend ---"
    cd frontend
    pnpm install --frozen-lockfile
    pnpm build
    cd "$PROJECT_DIR"

    # 3. Rebuild and restart
    echo "--- Rebuilding Docker images ---"
    docker compose -f docker-compose.prod.yml build
    docker compose -f docker-compose.prod.yml down
    docker compose -f docker-compose.prod.yml up -d

    # 4. Health check
    echo "--- Health check ---"
    sleep 10
    if curl -sf http://localhost/health > /dev/null 2>&1; then
      echo "Rollback successful. Health: $(curl -s http://localhost/health)"
    else
      echo "WARNING: Health check failed after rollback. Check logs."
      docker compose -f docker-compose.prod.yml logs --tail=50 nestjs
    fi
    ```

    Make both scripts executable: `chmod +x deploy/deploy.sh deploy/rollback.sh`
