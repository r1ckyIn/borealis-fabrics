    **2. SoftDeleteToggle component** (per D-13, D-14):
    Create `frontend/src/components/common/SoftDeleteToggle.tsx`:
    ```typescript
    import { Switch, Space } from 'antd';
    import { useUser } from '@/store/authStore';

    export interface SoftDeleteToggleProps {
      showDeleted: boolean;
      onChange: (checked: boolean) => void;
    }

    /**
     * Toggle switch for showing/hiding soft-deleted records.
     * Only visible to admin users (boss/developer).
     */
    export function SoftDeleteToggle({ showDeleted, onChange }: SoftDeleteToggleProps) {
      const user = useUser();

      if (!user?.isAdmin) return null;

      return (
        <Space>
          <Switch
            checkedChildren="显示已删除"
            unCheckedChildren="仅显示活跃"
            checked={showDeleted}
            onChange={onChange}
          />
        </Space>
      );
    }
    ```
