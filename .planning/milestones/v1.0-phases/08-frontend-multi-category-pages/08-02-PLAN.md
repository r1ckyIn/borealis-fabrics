```typescript
interface SpecFieldConfig {

  label: string;

  placeholder?: string;
}

const SPEC_FIELDS: Record<string, SpecFieldConfig[]> = {
  IRON_FRAME: [

    { name: 'specification', label: '规格', type: 'text', placeholder: '如 3人位 L型' },
  ],
  MOTOR: [
    { name: 'modelNumber', label: '型号', type: 'text', required: true },
    { name: 'specification', label: '详细规格', type: 'text' },

  MATTRESS: [
    { name: 'specification', label: '规格尺寸', type: 'text', required: true, placeholder: '如 190x140x11cm' },
  ],
  ACCESSORY: [
    { name: 'modelNumber', label: '型号', type: 'text' },
    { name: 'specification', label: '规格', type: 'text' },
  ],
};
```

7. Render common fields: 产品名称 (required), 默认单价 (InputNumber, optional), 备注 (TextArea, optional).

8. Render dynamic spec fields by iterating `SPEC_FIELDS[subCategory]`:
```typescript
function renderSpecField(field: SpecFieldConfig) {
  const rules = field.required ? [{ required: true, message: `请输入${field.label}` }] : [];
  switch (field.type) {
    case 'text': return <Input placeholder={field.placeholder || `请输入${field.label}`} />;
    case 'number': return <InputNumber style={{ width: '100%' }} />;
    case 'select': return <Select options={field.options} />;
  }
}
```

   - For edit: call `updateProduct(id, data)`, navigate to detail page on success

10. Page title: `${isEdit ? '编辑' : '新增'}${PRODUCT_SUB_CATEGORY_LABELS[subCategory]}`.

11. Breadcrumb: back link to `/products/${category}`.

12. Default export for lazy loading.
